# Blockchain/Web3 Patterns

Web3 applications require specialized patterns for wallet connections, transaction handling, token displays, and blockchain interactions. This guide covers essential components and patterns for building secure and user-friendly DeFi applications.

## Core Web3 Components

### Wallet Connection

The foundation of Web3 applications is wallet connectivity:

```tsx
import { WalletConnect, WalletButton, WalletStatus } from '@/components/web3'

export function Web3Dashboard() {
  const { address, isConnected, connect, disconnect } = useWallet()

  return (
    <div className="space-y-6">
      <WalletConnect>
        {!isConnected ? (
          <WalletButton onClick={connect}>
            Connect Wallet
          </WalletButton>
        ) : (
          <WalletStatus 
            address={address}
            onDisconnect={disconnect}
          />
        )}
      </WalletConnect>
      
      {isConnected && (
        <div className="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-3 gap-6">
          <TokenBalance address={address} />
          <TransactionHistory address={address} />
          <NetworkStatus />
        </div>
      )}
    </div>
  )
}
```

### Multi-Wallet Support

Support multiple wallet providers:

```tsx
export function WalletConnectModal({ open, onOpenChange }: WalletConnectModalProps) {
  const walletOptions = [
    { name: 'MetaMask', icon: '/metamask.svg', connector: 'metamask' },
    { name: 'WalletConnect', icon: '/walletconnect.svg', connector: 'walletconnect' },
    { name: 'Coinbase Wallet', icon: '/coinbase.svg', connector: 'coinbase' },
  ]

  return (
    <Dialog open={open} onOpenChange={onOpenChange}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Connect Wallet</DialogTitle>
          <DialogDescription>
            Choose your preferred wallet to connect to the application
          </DialogDescription>
        </DialogHeader>
        <div className="grid gap-3">
          {walletOptions.map((wallet) => (
            <Button
              key={wallet.connector}
              variant="outline"
              className="justify-start h-auto p-4"
              onClick={() => connect(wallet.connector)}
            >
              <img 
                src={wallet.icon} 
                alt={wallet.name}
                className="w-6 h-6 mr-3"
              />
              <div className="text-left">
                <div className="font-medium">{wallet.name}</div>
                <div className="text-sm text-muted-foreground">
                  Connect using {wallet.name}
                </div>
              </div>
            </Button>
          ))}
        </div>
      </DialogContent>
    </Dialog>
  )
}
```

## Transaction Patterns

### Transaction Status Component

```tsx
export function TransactionStatus({ transaction }: TransactionStatusProps) {
  const getStatusColor = (status: TransactionStatus) => {
    switch (status) {
      case 'pending': return 'text-yellow-600'
      case 'success': return 'text-green-600'
      case 'failed': return 'text-red-600'
      default: return 'text-muted-foreground'
    }
  }

  return (
    <Card>
      <CardContent className="p-4">
        <div className="flex items-center justify-between">
          <div className="flex items-center gap-3">
            <div className={getStatusColor(transaction.status)}>
              {transaction.status === 'pending' && <Loader2 className="h-4 w-4 animate-spin" />}
              {transaction.status === 'success' && <CheckCircle className="h-4 w-4" />}
              {transaction.status === 'failed' && <XCircle className="h-4 w-4" />}
            </div>
            <div>
              <div className="font-medium">{transaction.type}</div>
              <div className="text-sm text-muted-foreground">
                {transaction.hash.slice(0, 10)}...
              </div>
            </div>
          </div>
          <Button size="sm" variant="ghost" asChild>
            <a 
              href={`https://etherscan.io/tx/${transaction.hash}`}
              target="_blank"
              rel="noopener noreferrer"
            >
              <ExternalLink className="h-4 w-4" />
            </a>
          </Button>
        </div>
      </CardContent>
    </Card>
  )
}
```

## DeFi Components

### Swap Interface

```tsx
export function SwapInterface() {
  const [fromToken, setFromToken] = useState<Token | null>(null)
  const [toToken, setToToken] = useState<Token | null>(null)
  const [fromAmount, setFromAmount] = useState('')

  const handleSwap = async () => {
    if (!fromToken || !toToken || !fromAmount) return
    
    try {
      const tx = await executeSwap({
        fromToken,
        toToken,
        amount: fromAmount,
        slippage: 0.5,
      })
      
      toast.success('Swap initiated successfully!')
    } catch (error) {
      toast.error('Swap failed. Please try again.')
    }
  }

  return (
    <Card className="w-full max-w-md mx-auto">
      <CardHeader>
        <CardTitle>Swap Tokens</CardTitle>
      </CardHeader>
      <CardContent className="space-y-4">
        <div className="space-y-2">
          <Label>From</Label>
          <div className="flex gap-2">
            <Input
              type="number"
              placeholder="0.0"
              value={fromAmount}
              onChange={(e) => setFromAmount(e.target.value)}
              className="flex-1"
            />
            <TokenSelector
              selectedToken={fromToken}
              onTokenSelect={setFromToken}
            />
          </div>
        </div>
        
        <Button
          onClick={handleSwap}
          disabled={!fromToken || !toToken || !fromAmount}
          className="w-full"
        >
          Swap Tokens
        </Button>
      </CardContent>
    </Card>
  )
}
```

### Transaction Confirmation

```tsx
export function TransactionConfirmation({ 
  transaction, 
  onConfirm, 
  onCancel 
}: TransactionConfirmationProps) {
  return (
    <Dialog open onOpenChange={onCancel}>
      <DialogContent>
        <DialogHeader>
          <DialogTitle>Confirm Transaction</DialogTitle>
          <DialogDescription>
            Please review the transaction details before confirming
          </DialogDescription>
        </DialogHeader>
        
        <div className="space-y-4">
          <div className="p-4 border rounded-lg space-y-2">
            <div className="flex justify-between">
              <span className="text-muted-foreground">Type:</span>
              <span className="font-medium">{transaction.type}</span>
            </div>
            <div className="flex justify-between">
              <span className="text-muted-foreground">Amount:</span>
              <span className="font-medium">{transaction.amount} {transaction.symbol}</span>
            </div>
            <div className="flex justify-between">
              <span className="text-muted-foreground">Gas Fee:</span>
              <span className="font-medium">{transaction.gasFee} ETH</span>
            </div>
          </div>
        </div>
        
        <DialogFooter>
          <Button variant="outline" onClick={onCancel}>
            Cancel
          </Button>
          <Button onClick={onConfirm}>
            Confirm Transaction
          </Button>
        </DialogFooter>
      </DialogContent>
    </Dialog>
  )
}
```

## Best Practices

### Error Handling
- Always handle network errors gracefully
- Provide clear error messages for failed transactions
- Implement retry mechanisms for temporary failures

### User Experience
- Show loading states during blockchain interactions
- Provide transaction status updates
- Use progressive disclosure for complex operations

### Security
- Validate all user inputs
- Use secure connection methods
- Implement proper access controls

These patterns provide a solid foundation for building secure and user-friendly Web3 applications.
